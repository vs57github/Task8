n = int(input("Введите количество строк: "))
m = int(input("Введите количество столбцов: "))
k = int(input("Введите количество долек, которое нужно отломить: "))

if k > n * m:
    print("Невозможно отломить такое количество долек")
elif k % n == 0 or k % m == 0:
    print("Да, можно отломить ровно", k, "долек")
else:
    print("Нет, нельзя отломить ровно", k, "долек")
